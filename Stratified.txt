model{

## likelihood/sampling model for the basket trials:
	for(i in 1:Nobs){

		y[i] ~ dnorm(m.y[i], tau.y)
		m.y[i] <- z1[i]*gamma1 + z2[i]*gamma2 + Trt[i]*theta

	}
	
		# priors
		gamma1 ~ dnorm(0, 0.1)
		gamma2 ~ dnorm(0, 0.1)

		theta ~ dnorm(Prior.theta, prec.theta)
		prec.theta <- pow(Prior.sd.theta, -2)

		tau.y ~ dgamma(0.5, 0.005)	

		pCat[1] <- 1 - step(sig.theta[1] - theta)
		pCat[2] <- 1 - step(sig.theta[2] - theta)
		
}
